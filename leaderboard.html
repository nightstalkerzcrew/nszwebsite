const CONFIG = { SHEETS: { SUBMISSIONS: "Einreichungen", LEADERBOARD: "Leaderboard", TRACKS: "Strecken", CARS: "Autos", DRIVERS: "Fahrer" }, STATUS: { OPEN: "Offen", ACCEPTED: "Akzeptiert", REJECTED: "Abgelehnt" }, SUBMISSION_COLUMNS: { TIMESTAMP: 1, DISCORD_ID: 2, DRIVER_NAME: 3, TRACK: 4, CAR: 5, TIME: 6, PROOF: 7, STATUS: 8, COMMENT: 9 } }; function onOpen() { SpreadsheetApp.getUi() .createMenu("Leaderboard Tools") .addItem("Alles aktualisieren", "updateSystem") .addItem("Dropdowns einrichten", "setupValidation") .addItem("Leaderboard aktualisieren", "updateLeaderboard") .addToUi(); } function onEdit(e) { if (!e || !e.range) return; try { const sheet = e.range.getSheet(); if (sheet.getName() !== CONFIG.SHEETS.SUBMISSIONS) return; const row = e.range.getRow(); const col = e.range.getColumn(); if (row === 1) return; const relevantColumns = [ CONFIG.SUBMISSION_COLUMNS.DISCORD_ID, CONFIG.SUBMISSION_COLUMNS.DRIVER_NAME, CONFIG.SUBMISSION_COLUMNS.TRACK, CONFIG.SUBMISSION_COLUMNS.CAR, CONFIG.SUBMISSION_COLUMNS.TIME, CONFIG.SUBMISSION_COLUMNS.STATUS ]; if (!relevantColumns.includes(col)) return; updateSystem(); } catch (error) { Logger.log("onEdit Fehler: " + error.message); } } function updateSystem() { const lock = LockService.getDocumentLock(); try { lock.waitLock(30000); updateDriversSheet(); updateCarsSheet(); updateTracksSheet(); updateLeaderboard(); setupValidation(); } catch (error) { Logger.log("updateSystem Fehler: " + error.message); throw error; } finally { lock.releaseLock(); } } function updateDriversSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const inputSheet = ss.getSheetByName(CONFIG.SHEETS.SUBMISSIONS); const driverSheet = ss.getSheetByName(CONFIG.SHEETS.DRIVERS); if (!inputSheet || !driverSheet) { throw new Error('Blätter "Einreichungen" oder "Fahrer" nicht gefunden.'); } const data = inputSheet.getDataRange().getValues(); driverSheet.clear(); driverSheet.getRange(1, 1, 1, 2).setValues([["Discord-ID", "Fahrername"]]); if (data.length < 2) { formatSimpleSheet(driverSheet, 2, 0); return; } const driversMap = {}; for (let i = 1; i < data.length; i++) { const discordId = String(data[i][CONFIG.SUBMISSION_COLUMNS.DISCORD_ID - 1] || "").trim(); const driverName = String(data[i][CONFIG.SUBMISSION_COLUMNS.DRIVER_NAME - 1] || "").trim(); if (!discordId || !driverName) continue; if (!driversMap[discordId]) { driversMap[discordId] = driverName; } } const drivers = Object.keys(driversMap) .map(id => [id, driversMap[id]]) .sort((a, b) => a[1].localeCompare(b[1], "de")); if (drivers.length > 0) { driverSheet.getRange(2, 1, drivers.length, 2).setValues(drivers); } formatSimpleSheet(driverSheet, 2, drivers.length); } function updateCarsSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const inputSheet = ss.getSheetByName(CONFIG.SHEETS.SUBMISSIONS); const carSheet = ss.getSheetByName(CONFIG.SHEETS.CARS); if (!inputSheet || !carSheet) { throw new Error('Blätter "Einreichungen" oder "Autos" nicht gefunden.'); } const data = inputSheet.getDataRange().getValues(); const existingCars = getExistingSingleColumnValues(carSheet); for (let i = 1; i < data.length; i++) { const car = String(data[i][CONFIG.SUBMISSION_COLUMNS.CAR - 1] || "").trim(); if (car) existingCars.add(car); } const cars = Array.from(existingCars) .sort((a, b) => a.localeCompare(b, "de")) .map(car => [car]); carSheet.clear(); carSheet.getRange(1, 1, 1, 1).setValues([["Auto"]]); if (cars.length > 0) { carSheet.getRange(2, 1, cars.length, 1).setValues(cars); } formatSimpleSheet(carSheet, 1, cars.length); } function updateTracksSheet() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const inputSheet = ss.getSheetByName(CONFIG.SHEETS.SUBMISSIONS); const trackSheet = ss.getSheetByName(CONFIG.SHEETS.TRACKS); if (!inputSheet || !trackSheet) { throw new Error('Blätter "Einreichungen" oder "Strecken" nicht gefunden.'); } const data = inputSheet.getDataRange().getValues(); const existingTrackTypes = getExistingTrackTypes(trackSheet); const allTracks = new Set(Object.keys(existingTrackTypes)); for (let i = 1; i < data.length; i++) { const track = String(data[i][CONFIG.SUBMISSION_COLUMNS.TRACK - 1] || "").trim(); if (track) allTracks.add(track); } const tracks = Array.from(allTracks) .sort((a, b) => a.localeCompare(b, "de")) .map(track => [track, existingTrackTypes[track] || ""]); trackSheet.clear(); trackSheet.getRange(1, 1, 1, 2).setValues([["Strecke", "Renntyp"]]); if (tracks.length > 0) { trackSheet.getRange(2, 1, tracks.length, 2).setValues(tracks); } formatSimpleSheet(trackSheet, 2, tracks.length); } function updateLeaderboard() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const inputSheet = ss.getSheetByName(CONFIG.SHEETS.SUBMISSIONS); const leaderboardSheet = ss.getSheetByName(CONFIG.SHEETS.LEADERBOARD); if (!inputSheet || !leaderboardSheet) { throw new Error('Blätter "Einreichungen" oder "Leaderboard" nicht gefunden.'); } const data = inputSheet.getDataRange().getValues(); leaderboardSheet.clear(); leaderboardSheet.getRange(1, 1, 1, 5).setValues([["Rang", "Fahrer", "Strecke", "Fahrzeug", "Beste Zeit"]]); if (data.length < 2) { formatLeaderboardSheet(leaderboardSheet, 0, 5); return; } const bestTimes = {}; for (let i = 1; i < data.length; i++) { const driver = String(data[i][CONFIG.SUBMISSION_COLUMNS.DRIVER_NAME - 1] || "").trim(); const track = String(data[i][CONFIG.SUBMISSION_COLUMNS.TRACK - 1] || "").trim(); const car = String(data[i][CONFIG.SUBMISSION_COLUMNS.CAR - 1] || "").trim(); const rawTime = data[i][CONFIG.SUBMISSION_COLUMNS.TIME - 1]; const status = normalizeStatus(data[i][CONFIG.SUBMISSION_COLUMNS.STATUS - 1]); if (!driver || !track || !car || rawTime === "" || rawTime === null) continue; if (status !== CONFIG.STATUS.ACCEPTED.toLowerCase()) continue; const timeMs = parseTimeToMilliseconds(rawTime); if (timeMs === null) continue; const key = driver + "||" + track; if (!bestTimes[key] || timeMs < bestTimes[key].timeMs) { bestTimes[key] = { driver: driver, track: track, car: car, timeMs: timeMs }; } } const leaderboard = Object.values(bestTimes) .sort((a, b) => a.timeMs - b.timeMs) .map((entry, index) => [ index + 1, entry.driver, entry.track, entry.car, formatMilliseconds(entry.timeMs) ]); if (leaderboard.length > 0) { leaderboardSheet.getRange(2, 1, leaderboard.length, 5).setValues(leaderboard); } formatLeaderboardSheet(leaderboardSheet, leaderboard.length, 5); } function setupValidation() { const ss = SpreadsheetApp.getActiveSpreadsheet(); const inputSheet = ss.getSheetByName(CONFIG.SHEETS.SUBMISSIONS); const driverSheet = ss.getSheetByName(CONFIG.SHEETS.DRIVERS); const trackSheet = ss.getSheetByName(CONFIG.SHEETS.TRACKS); const carSheet = ss.getSheetByName(CONFIG.SHEETS.CARS); if (!inputSheet || !driverSheet || !trackSheet || !carSheet) { throw new Error("Ein oder mehrere Blätter fehlen."); } const maxRows = Math.max(inputSheet.getMaxRows(), 2); const driverLastRow = Math.max(driverSheet.getLastRow(), 2); const trackLastRow = Math.max(trackSheet.getLastRow(), 2); const carLastRow = Math.max(carSheet.getLastRow(), 2); const driverRule = SpreadsheetApp.newDataValidation() .requireValueInRange(driverSheet.getRange(2, 2, driverLastRow - 1, 1), true) .setAllowInvalid(true) .build(); const trackRule = SpreadsheetApp.newDataValidation() .requireValueInRange(trackSheet.getRange(2, 1, trackLastRow - 1, 1), true) .setAllowInvalid(true) .build(); const carRule = SpreadsheetApp.newDataValidation() .requireValueInRange(carSheet.getRange(2, 1, carLastRow - 1, 1), true) .setAllowInvalid(true) .build(); const statusRule = SpreadsheetApp.newDataValidation() .requireValueInList( [CONFIG.STATUS.OPEN, CONFIG.STATUS.ACCEPTED, CONFIG.STATUS.REJECTED], true ) .setAllowInvalid(false) .build(); inputSheet.getRange(2, CONFIG.SUBMISSION_COLUMNS.DRIVER_NAME, maxRows - 1, 1).setDataValidation(driverRule); inputSheet.getRange(2, CONFIG.SUBMISSION_COLUMNS.TRACK, maxRows - 1, 1).setDataValidation(trackRule); inputSheet.getRange(2, CONFIG.SUBMISSION_COLUMNS.CAR, maxRows - 1, 1).setDataValidation(carRule); inputSheet.getRange(2, CONFIG.SUBMISSION_COLUMNS.STATUS, maxRows - 1, 1).setDataValidation(statusRule); } function getExistingSingleColumnValues(sheet) { const lastRow = sheet.getLastRow(); const values = new Set(); if (lastRow < 2) return values; const data = sheet.getRange(2, 1, lastRow - 1, 1).getValues(); for (let i = 0; i < data.length; i++) { const value = String(data[i][0] || "").trim(); if (value) values.add(value); } return values; } function getExistingTrackTypes(sheet) { const lastRow = sheet.getLastRow(); const map = {}; if (lastRow < 2) return map; const data = sheet.getRange(2, 1, lastRow - 1, 2).getValues(); for (let i = 0; i < data.length; i++) { const track = String(data[i][0] || "").trim(); const type = String(data[i][1] || "").trim(); if (track) { map[track] = type; } } return map; } function normalizeStatus(value) { return String(value || "").trim().toLowerCase(); } function parseTimeToMilliseconds(timeValue) { if (timeValue instanceof Date) { return ( timeValue.getHours() * 3600000 + timeValue.getMinutes() * 60000 + timeValue.getSeconds() * 1000 + timeValue.getMilliseconds() ); } const str = String(timeValue).trim(); let match = str.match(/^(\d+):(\d{1,2})\.(\d{1,3})$/); if (match) { const minutes = parseInt(match[1], 10); const seconds = parseInt(match[2], 10); const millis = parseInt(match[3].padEnd(3, "0"), 10); return minutes * 60000 + seconds * 1000 + millis; } match = str.match(/^(\d+):(\d{1,2}):(\d{1,3})$/); if (match) { const minutes = parseInt(match[1], 10); const seconds = parseInt(match[2], 10); const millis = parseInt(match[3].padEnd(3, "0"), 10); return minutes * 60000 + seconds * 1000 + millis; } match = str.match(/^(\d+):(\d{1,2})$/); if (match) { const minutes = parseInt(match[1], 10); const seconds = parseInt(match[2], 10); return minutes * 60000 + seconds * 1000; } if (!isNaN(str) && str !== "") { return Math.round(parseFloat(str) * 24 * 60 * 60 * 1000); } return null; } function formatMilliseconds(ms) { const minutes = Math.floor(ms / 60000); const seconds = Math.floor((ms % 60000) / 1000); const millis = ms % 1000; return `${minutes}:${String(seconds).padStart(2, "0")}.${String(millis).padStart(3, "0")}`; } function formatSimpleSheet(sheet, colCount, rowCount) { sheet.setFrozenRows(1); sheet.autoResizeColumns(1, colCount); const headerRange = sheet.getRange(1, 1, 1, colCount); headerRange.setFontWeight("bold"); headerRange.setBackground("#111827"); headerRange.setFontColor("#ffffff"); headerRange.setHorizontalAlignment("center"); if (rowCount > 0) { const dataRange = sheet.getRange(2, 1, rowCount, colCount); dataRange.setHorizontalAlignment("center"); dataRange.setVerticalAlignment("middle"); } } function formatLeaderboardSheet(sheet, rowCount, colCount) { sheet.setFrozenRows(1); sheet.autoResizeColumns(1, colCount); const headerRange = sheet.getRange(1, 1, 1, colCount); headerRange.setFontWeight("bold"); headerRange.setBackground("#111827"); headerRange.setFontColor("#ffffff"); headerRange.setHorizontalAlignment("center"); if (rowCount > 0) { const dataRange = sheet.getRange(2, 1, rowCount, colCount); dataRange.setHorizontalAlignment("center"); dataRange.setVerticalAlignment("middle"); } if (rowCount >= 1) { sheet.getRange(2, 1, 1, colCount).setBackground("#ffd700").setFontWeight("bold"); } if (rowCount >= 2) { sheet.getRange(3, 1, 1, colCount).setBackground("#c0c0c0").setFontWeight("bold"); } if (rowCount >= 3) { sheet.getRange(4, 1, 1, colCount).setBackground("#cd7f32").setFontWeight("bold"); } }
